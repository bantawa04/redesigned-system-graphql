Posts

Get all posts

const  graphqlQuery  = {
    query:  `
    {
        getPosts{
            posts {
                id
                title
                content
                image
                author_id
                created_at
                updated_at
            }
        }
    }`
};
fetch("https://json-graphql-endpoint.herokuapp.com/graphql", {
    method:  "POST",
    headers: {
    Accept:  "application/json",
        "Content-Type":  "application/json",
    },
    body:  JSON.stringify(graphqlQuery),
})
.then((response) =>  response.json())
.then((json) =>  console.log(json));
